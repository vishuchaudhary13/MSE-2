
# 1. Navigate to home directory
cd ~

# 2. Create index.html file
touch index.html

# 3. Add HTML content to the file
echo "<html><body><h1>Hello World</h1></body></html>" > index.html

# 4. Check current default permissions
ls -l index.html

# 5. Grant full permissions to all users
chmod 777 index.html

# Verify permissions
ls -l index.html



# 6. Change owner and group to www-data
sudo chown www-data:www-data index.html

# Verify ownership
ls -l index.html


# 7. Change permissions to 755
chmod 755 index.html

# 8. Display final permissions and ownership
ls -l index.html
